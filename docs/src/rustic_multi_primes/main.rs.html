use std::thread;
use num_cpus;

fn main() {
    let range = [2, 100000000];
    let divided_range = floor_divider(range);
    let mut comp_output: Vec<thread::JoinHandle<Vec<u128>>> = vec![];
    let mut results: Vec<u128> = vec![];
    //println!("{:?}", primes_test_10000(prime(0, 10000)));
    for pair in divided_range {
        let computation = thread::spawn(move || prime(pair.0, pair.1));
        comp_output.push(computation);
    }
    for handle in comp_output {
        results.append(&mut handle.join().unwrap());
    }
    results.sort();
    println!("{:?}", results);
    if primes_test_10000(results) {
        println!("Success!!!");
    }
}

fn floor_divider(range: [u128; 2]) -> Vec<(u128, u128)> {
    let cpu = num_cpus::get() - 1;
    let divider = (range[1] - range[0]) / cpu as u128;
    let mut new_range: Vec<u128> = vec![];
    let mut j = range[0];
    for _ in 0..cpu {
        j = j + divider;
        new_range.push(j);
    }
    println!("{:?}", new_range);
    let mut args_tuple: Vec<(u128, u128)> = vec![];
    let new_range_last = new_range.last().unwrap();
    if new_range_last + 1 >= range[1] {
        args_tuple.push((range[1], range[1]));
    } else {
        args_tuple.push((new_range_last + 1, range[1]));
    }
    args_tuple.push((range[0], new_range[0]));
    let new_range_len: usize = new_range.len().try_into().unwrap();
    for i in 2..(cpu + 1) {
        args_tuple.push((new_range[new_range_len - i] + 1, new_range[new_range_len - (i - 1)]));
    }
    println!("{:?}", args_tuple);
    return args_tuple
}

fn prime(first: u128, second: u128) -> Vec<u128> {
    let mut primes_vec: Vec<u128> = vec![];
    for i in first..(second + 1) {
        let mut not_prime = true;
        if i == 2 || i == 3 {
            not_prime = false;
        }
        let mut j: u128 = 2;
        while (j as f64) <= (i as f64).powf(0.5) {
            if i % j == 0 {
                not_prime = true;
                break;
            }
            not_prime = false;
            j += 1;
        }
        if !not_prime {
            primes_vec.push(i);
        }
    }
    return primes_vec
}
